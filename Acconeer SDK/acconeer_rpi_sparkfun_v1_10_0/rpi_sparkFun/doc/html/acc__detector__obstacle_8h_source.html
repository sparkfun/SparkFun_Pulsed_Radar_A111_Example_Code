54 /**
55  * @brief Activate an obstacle detector.
56  *
57  * Activate an obstacle detector associated by handle. Use acc_detector_obstacle_get_next
58  * to retrieve obstacles.
59  *
60  * Activating an already activated detector has no effect and true will be returned.
61  *
62  * If handle is a nullpointer the activation will fail and return false.
63  *
64  * @param[in] handle A handle for a detector to be activated
65  * @return True if activation succeeded otherwise false.
66  */

70 /**
71  * @brief Deactivate an obstacle detector.
72  *
73  * Deactivate an obstacle detector associated by the handle.
74  *
75  * Deactivating an already deactivated detector has no effect and true will be returned.
76  * If handle is a nullpointer the activation will fail and return false.
77  *
78  * If the deactivation fails other than if the handle is a nullpointer, the system is in
79  * an undefined state and further activation of detectors and services should be avoided
80  * in the same session.
81  *
82  * @param[in] handle A handle for a detector to be deactivated
83  * @return True if deactivation succeed otherwise false
84  *
85  */
86 bool acc_detector_obstacle_deactivate(acc_detector_obstacle_handle_t handle);
