31 /**
32  * @brief Background cancellation mode
33  *
34  * Coherent background: This method uses background cancellation per sweep, i.e., an average
35  * complex background sweep is substracted from each sweep collected before
36  * running the FFT.
37  * Non-coherent background: This method uses background cancellation on the calculated FFT
38  * envelope, i.e., an average FFT envelope is substracted from each
39  * running FFT.
40  *
41  * @note: The background estimation is performed in accordance with the selected cancellation mode
42  */
43 typedef enum
44 {
45  ACC_DETECTOR_OBSTACLE_CONFIGURATION_BACKGROUND_CANCELLATION_MODE_COHERENT = 0,
46  ACC_DETECTOR_OBSTACLE_CONFIGURATION_BACKGROUND_CANCELLATION_MODE_NON_COHERENT,
47 } acc_detector_obstacle_configuration_background_cancellation_mode_t;

50 /**
51  * @brief Threshold parameters for non coherent background cancellation
52  *
53  * @param stationary The amplitude limit for stationary objects close to the sensor
54  * @param moving The amplitude limit for moving objects and objects far from the sensor
55  * @param distance_limit_far For distances larger than the far limit, use the @ref moving threshold
56  * @param close_addition The amplitude increase at closest distance, applied for all velocities
57  * @param distance_limit_near @ref close_addition threshold is applied from first distance in the range and linearly falls off to 0 at first distance plus distance_limit_near
58  */
59 typedef struct
60 {
61  float stationary;
62  float moving;
63  float distance_limit_far;
64  float close_addition;
65  float distance_limit_near;
66 } acc_detector_obstacle_configuration_threshold_t;

445 /**
446  * @brief Set the edge detection level ratio for the obstacle distance calculation
447  *
448  * This configuration parameter controls the level used for edge detection for a found obstacle.
449  * The value should be in the range from 0 to 1. The obstacle distance is calculated from the
450  * near edge of the peak where the signal amplitude crosses a threshold. This threshold is given
451  * by the ratio parameter times the peak amplitude unless this product is lower than the detection
452  * threshold set by @ref acc_detector_obstacle_configuration_set_thresholds or
453  * @ref acc_detector_obstacle_configuration_set_thresholds. In the latter case,
454  * the detection threshold is used. A zero value of the ratio parameter ignores the peak level and
455  * the reported peak distance tells the position of the near edge of the peak where the amplitude
456  * crosses the peak detection threshold. A value of 1 cuts the edge-detection procedure short and
457  * reports the peak position.
458  *
459  * @param[in] obstacle_configuration The configuration that the setting shall be applied to
460  * @param edge_to_peak_ratio The value to be used for the edge to peak level ratio
461  */
462 void acc_detector_obstacle_configuration_set_edge_to_peak_ratio(acc_detector_obstacle_configuration_t obstacle_configuration,
463  float edge_to_peak_ratio);
